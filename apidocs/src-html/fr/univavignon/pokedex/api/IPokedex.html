001package fr.univavignon.pokedex.api;
002
003import java.util.Comparator;
004import java.util.List;
005
006/**
007 * IPokedex interface. An IPokedex aims to store all information about
008 * captured pokemon, as their default metadata as well.*
009 *
010 * @author fv
011 */
012public interface IPokedex extends IPokemonMetadataProvider, IPokemonFactory {
013
014    /**
015     * Returns the number of pokemon this pokedex contains.
016     *
017     * @return Number of pokemon in this pokedex.
018     */
019    int size();
020
021    /**
022     * Adds the given <tt>pokemon</tt> to this pokedex and returns
023     * it unique index.
024     *
025     * @param pokemon Pokemon to add to this pokedex.
026     * @return Index of this pokemon relative to this pokedex.
027     */
028    int addPokemon(Pokemon pokemon);
029
030    /**
031     * Locates the pokemon identified by the given <tt>id</tt>.
032     *
033     * @param id Unique pokedex relative identifier.
034     * @return Pokemon denoted by the given identifier.
035     * @throws PokedexException If the given <tt>index</tt> is not valid.
036     */
037    Pokemon getPokemon(int id) throws PokedexException;
038
039    /**
040     * Returns an unmodifiable list of all pokemons this pokedex contains.
041     *
042     * @return Unmodifiable list of all pokemons.
043     */
044    List<Pokemon> getPokemons();
045
046    /**
047     * Returns an unmodifiable list of all pokemons this pokedex contains.
048     * The list view will be sorted using the given <tt>order</tt>.
049     *
050     * @param order Comparator instance used for sorting the created view.
051     * @return Sorted unmodifiable list of all pokemons.
052     */
053    List<Pokemon> getPokemons(Comparator<Pokemon> order);
054
055}
