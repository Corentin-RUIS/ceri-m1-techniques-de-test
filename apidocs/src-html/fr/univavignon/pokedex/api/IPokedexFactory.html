001package fr.univavignon.pokedex.api;
002
003/**
004 * Factory interface for class that aims to create IPokedex instance.
005 *
006 * @author fv
007 */
008public interface IPokedexFactory {
009
010    /**
011     * Creates a new Pokédex instance using the given
012     * <code>metadataProvider</code> and <code>pokemonFactory</code>.
013     *
014     * @param metadataProvider Metadata provider the created Pokédex will use.
015     * @param pokemonFactory   Pokémon factory the created Pokédex will use.
016     * @return Created Pokédex instance.
017     */
018    IPokedex createPokedex(IPokemonMetadataProvider metadataProvider,
019                           IPokemonFactory pokemonFactory);
020
021}
