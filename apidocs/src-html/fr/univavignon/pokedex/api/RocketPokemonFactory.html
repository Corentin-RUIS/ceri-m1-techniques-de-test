001package fr.univavignon.pokedex.api;
002
003import java.util.HashMap;
004import java.util.Map;
005import java.util.Random;
006
007import org.apache.commons.collections4.map.UnmodifiableMap;
008
009import fr.univavignon.pokedex.api.IPokemonFactory;
010import fr.univavignon.pokedex.api.Pokemon;
011
012public class RocketPokemonFactory implements IPokemonFactory {
013
014    private static Map<Integer, String> index2name;
015
016    static {
017        Map<Integer, String> aMap = new HashMap<Integer, String>();
018        aMap.put(-1, "Ash's Pikachu");
019        aMap.put(0, "MISSINGNO");
020        aMap.put(1, "Bulbasaur");
021        //TODO : Gotta map them all !
022        index2name = UnmodifiableMap.unmodifiableMap(aMap);
023    }
024
025    private static int generateRandomStat() {
026        int total = 0;
027        for (int i = 0; i < 1000000; i++) {
028            Random rn = new Random();
029            int r = rn.nextInt(2);
030            total = total + r;
031        }
032        return total / 10000;
033    }
034
035    @Override
036    public Pokemon createPokemon(int index, int cp, int hp, int dust, int candy) {
037        String name;
038        if (!index2name.containsKey(index)) {
039            name = index2name.get(0);
040        } else {
041            name = index2name.get(index);
042        }
043        int attack;
044        int defense;
045        int stamina;
046        double iv;
047        if (index < 0) {
048            attack = 1000;
049            defense = 1000;
050            stamina = 1000;
051            iv = 0;
052        } else {
053            attack = RocketPokemonFactory.generateRandomStat();
054            defense = RocketPokemonFactory.generateRandomStat();
055            stamina = RocketPokemonFactory.generateRandomStat();
056            iv = 1;
057        }
058        return new Pokemon(index, name, attack, defense, stamina, cp, hp, dust, candy, iv);
059    }
060
061}
